296  output_.push(translationGrammar_->starting_symbol());

299  size_t ruleIndex;
300  switch (top.type()) {

302  if (token == Symbol::eof())
303  return;
304  else

306  break;
307  case Type::TERMINAL:

312  input_.pop();

314  } else {

316  ", expected " + top.name() + ".");
317  }
318  break;
319  case Type::NONTERMINAL:
320  ruleIndex = llTable_.rule_index(top, token);
321  if (ruleIndex < translationGrammar_->rules().size()) {
322  auto &rule = translationGrammar_->rules()[ruleIndex];
323 
324  auto obegin = output_.replace(top, rule.output());
325  input_.replace(input_.begin(), rule.input());
326  create_attibute_actions(obegin, rule.actions(), attributeActions);
327  } else {

329  }
